Pseudocode

MergeSort(arr, l, r):
  if l >= r: return
  m = (l + r) / 2
  MergeSort(arr, l, m)
  MergeSort(arr, m+1, r)
  Merge(arr, l, m, r)

Merge(arr, l, m, r):
  create left[] and right[]
  while elements remain in both:
    compare and place smaller into arr
  copy remaining elements
